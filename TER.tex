		\subsubsection{Variétés différentielles:}

		Soit $\mathcal{M}$ un espace topologique séparé, soit $m\in \N$ et soit $r\in \N \cup \{\infty\}$.

		\begin{defi}[Carte]
		Une carte sur $\mathcal{M}$ à valeur dans $\R ^m$ est la donnée d'un couple $(U, \phi)$ où $U$ est un ouvert de $\mathcal{M}$ et $\phi : U \to \phi(U) \in \R^m$ est un homéomorphisme de $U$ sur son image avec $\phi(U)$ ouvert dans $\R^m$.
		\end{defi}

		\begin{defi}[Atlas]
		Un $\mathscr{C}^r$-atlas de $\mathcal{M}$ dans $\R^m$ est un ensemble $\mathscr{A}$ de cartes $(U, \phi)$ à valeur dans $\R^m$ dont les domaines de cartes $U$ recouvrent $\mathcal{M}$ et qui sont $\mathscr{C}^r$-compatibles, c'est à dire telles que:
		$$\forall (U, \phi), (V, \psi) \in \mathscr{A}, \quad 
		\psi \circ \phi ^{-1} : \phi(U\cap V) \to \psi(U \cap V) \quad \text{est un }\mathscr{C}^r\text{-difféomorphisme}$$
		Deux $\mathscr{C}^r$-atlas sont $\mathscr{C}^r$-compatibles si leur réunion est un $\mathscr{C}^r$-atlas. Cela définit une relation d'équivalence et la réunion de la classe d'équivalence d'un $\mathscr{C}^r$-atlas est l'unique $\mathscr{C}^r$-atlas maximal qui le contient.
		\end{defi}

		\begin{defi}[Variété différentielle]
		Une variété différentielle de classe $\mathscr{C}^r$ (ou lisse si $r = \infty$) de dimension $m$ est un espace topologique séparé et à base dénombrable d'ouverts muni d'un $\mathscr{C}^r$-atlas maximal à valeur dans $\R^m$.
		\end{defi}

		\begin{defi}
		Soient $\mathcal{M}$, $\mathcal{N}$ deux variétés de classe $\mathscr{C}^r$ et $f: \mathcal{M} \to \mathcal{N}$. On dira que $f$ est $\mathscr{C}^r$ (ou lisse si $r = \infty$) si $\forall x\in \mathcal{M}$ il existe une carte $(U, \phi)$ de $\mathcal{M}$ telle que $x\in U$ et une carte $(V, \psi)$ de $\mathcal{N}$ telle que $f(x) \in V$ et telles que $f(U)\subset V$ et si l'application $f$ lue dans les cartes locales 
		$$\psi \circ f \circ \phi^{-1}:\phi(U)\to \psi(V)$$
		est $\mathscr{C}^r$ en $\phi(x)$. \\
		Par composition des applications différentiables, cette définition ne dépend pas du choix des cartes.
		\end{defi}

		\begin{pro}
		Soient $\mathcal{M}$, $\mathcal{N}$ et $\mathcal{P}$ des variétés $\mathscr{C}^r$ et $f:\mathcal{M}\to \mathcal{N}$ et $g:\mathcal{N}\to\mathcal{P}$ des applications $\mathscr{C}^r$. Alors $g \circ f: \mathcal{M} \to \mathcal{P}$ est $\mathscr{C}^r$.
		\end{pro}
		\begin{preu}
		Par composition des applications différentiables
		\end{preu}

		\begin{defi}[Sous-variété]
		Soit $\mathcal{M}$ une variété lisse de dimension $m$ et soit $\mathcal{N}$ une partie de $\mathcal{M}$ telle qu'il existe $n\leq m$ tel que pour toute carte $(U, \phi)$ de $\mathcal{M}$ telle que $U \cap \mathcal{N} \neq \emptyset$ on ait
		$$\phi : U\cap \mathcal{N} \to \phi(U)\cap \R^n$$
		est un homéomorphisme.\\
		Dans ce cas on dit que $\mathcal{N}$ est une sous-variété lisse de $\mathcal{M}$ de dimension $n$ et de codimension $m-n$ dans $\mathcal{M}$.
		\end{defi}

		\begin{pro}
		Soit $\mathcal{N}$ une sous variété lisse de $\mathcal{M}$ alors si on note $\{(U_\alpha, \phi_\alpha), \alpha \in A\}$ l'atlas lisse de $\mathcal{M}$ alors l'ensemble $\{(U_\alpha \cap \mathcal{N}, \phi_\alpha), \alpha \in A\}$ est un atlas lisse sur $\mathcal{N}$ qui fait de $\mathcal{N}$ une variété lisse de dimension $n$. \\
		De plus, l'inclusion $\mathcal{N} \to \mathcal{M}$ est lisse.
		\end{pro}

		\subsubsection{Fibré tangent:}
		Soit $U\subset \R^m$ un ouvert, on pose $TU = U\times\R^m$ et on note 
		$$\tau_U : TU \to U, (x, v) \mapsto x$$
		Aussi, pour $x\in U$ on note $T_xU = \tau_U^{-1}({x})$ la fibre au dessus de $x$ qu'on muni d'une structure d'espace vectoriel via
		$$(x,v)+(x,v')=(x, v+v')$$
		$$\lambda(x, v) = (x, \lambda v)$$
		Ainsi $T_xU$ est canoniquement isomorphe à $\R^m$ via $(x, v) \mapsto v$. \\

		Soit $U\subset \R^m$, $V\subset \R^n$ des ouverts et $f:U\to V$ lisse, on définit l'application tangente de $f$ comme
		$$Tf: TU \to TV, (x,v) \mapsto (f(x), df_x.v)$$
		Alors, $T_xf = Tf|_{T_xU} : T_xU \to T_{f(x)}V$ est une application linéaire.\\

		Soit $\mathcal{M}$ une variété lisse de dimension $n$ et soit $\mathscr{A} = \{(U_\alpha, \phi_\alpha), \alpha \in A\}$ son atlas, notons $V_\alpha = \phi_\alpha(U_\alpha) \subset \R^m$, $V_{\beta\alpha} = \phi_\alpha(U_\alpha \cap U_\beta) \subset \R^m$ et $\varphi_{\beta\alpha} = \phi_\beta \circ \phi_\alpha^{-1} : V_{\beta\alpha} \to V_{\alpha\beta}$ les fonctions de transitions. Notons $W_\alpha = TV_\alpha = V_\alpha \times \R^m \subset \R^{2m}$ et $W_{\beta\alpha} = TV_{\beta\alpha} = V_{\beta\alpha} \times \R^m \subset \R^{2m}$. Soit alors $\psi_{\beta\alpha} = T\varphi_{\beta\alpha}: W_{\beta\alpha}\to W_{\alpha\beta}$. \\

		On définit alors une relation binaire sur $X = \DUnion _{\alpha \in A} W_\alpha$ pour $(x, u)\in W_{\beta\alpha} \subset W_\alpha$ et  $(y, v) \in W_{\alpha\beta} \subset W_\beta$,
		$$(x,u) \sim (y,v) \iff (y,v) = \psi_{\beta\alpha}(x,u)$$ 
		C'est une relation d'équivalence: elle est reflexive car $\psi_{\alpha\alpha} = id_{W_\alpha}$, symétrique car $\psi_{\beta\alpha}^{-1} = \psi_{\alpha\beta}$ et transitive car $W_{\beta\alpha} \cap \psi_{\beta\alpha}^{-1}(W_{\gamma\beta}) \subset W_{\gamma\alpha}$ et $\psi_{\gamma\beta} \circ \psi_{\beta\alpha} = \psi_{\gamma\alpha}$ sur $W_{\beta\alpha} \cap \psi_{\beta\alpha}^{-1}(W_{\gamma\beta})$. \\

		On définit alors $T\mathcal{M}$ comme l'ensemble quotient $X / \sim$. On munie $X$ de la topologie union disjointe et $T\mathcal{M}$ de la topologie quotient. \\

		Considérons les applications $f_\alpha : W_\alpha \to T\mathcal{M}$ qui envoie les éléments de $W_\alpha$ dans leur classe d'équivalence dans $T\mathcal{M}$. Alors ce sont des homéomorphismes sur leur image. En effet, par définitions de la topologie union disjointe les applications $W_\alpha \to X$ sont des homéomorphismes sur leur image. Ensuite si $x,y\in W_\alpha$ tels que $x\sim y$ alors $x = y$. Donc ces applications descendent au quotient en les applications $f_\alpha$ qui sont donc des homéomorphismes sur leur image. Ainsi, notons $\tilde{W_\alpha}$ l'image de $f_\alpha$ et posons $\tilde{\phi_\alpha} = f_\alpha^{-1} : \tilde{W_\alpha} \to W_\alpha$. \\

		Alors l'ensemble $\{(\tilde{W_\alpha}, \tilde{\phi_\alpha}), \alpha \in A\}$ est un atlas lisse sur $T\mathcal{M}$. En effet, par construction on a $\tilde{\phi_\beta} \circ \tilde{\phi_\alpha}^{-1} = \psi_{\beta\alpha} = T\varphi_{\beta\alpha}$ qui sont lisses. \\

		De plus, $T\mathcal{M}$ est séparé. En effet, considérons les applications continues $\tau_\alpha : W_\alpha \to \mathcal{M}, (x, v) \mapsto \phi_\alpha^{-1}(x)$. Elles induisent une application continue $\tau : X \to \mathcal{M}$. Alors pour $(x, u) \in W_\alpha$ et $(y, v) \in W_\beta$ tels que $(x,u)\sim (y, v)$ alors en particulier $y = \varphi_{\beta\alpha}(x) = \phi_\beta \circ \phi_\alpha^{-1} (x)$, donc $\tau(x,u) = \tau(y, v)$. Ainsi, comme $\mathcal{M}$ est séparé, $T\mathcal{M}$ l'est aussi. \\

		Finalement, $T\mathcal{M}$ muni de l'atlas lisse $\{(\tilde{W_\alpha}, \tilde{\phi_\alpha}), \alpha \in A\}$ est une variété lisse de dimension $2m$. \\

		Notons $\tau_\mathcal{M} : T\mathcal{M} \to \mathcal{M}$, l'application $\tau$ descendue au quotient qui est clairement lisse par construction. Notons $T_x\mathcal{M} = \tau_\mathcal{M}^{-1}(x)$ la fibre au dessus de $x$. Soit $(U\alpha, \phi_\alpha)$ une carte autour de $x$ dans $\mathcal{M}$ et soit $(\tilde{W_\alpha}, \tilde{\phi_\alpha})$ la carte associée dans $T\mathcal{M}$, alors $\tilde{\phi_\alpha} : \{x\}\times T_x\mathcal{M} \to \{\phi_\alpha(x)\}\times \R^n$ est une bijection. On muni $T_x\mathcal{M}$ d'une structure d'espace vectoriel tel que cette bijection devienne une application linéaire. Cette définition ne dépend pas du choix des cartes puisque les fonctions de transitions de $T\mathcal{M}$ sont des applications tangentes donc sont linéaires. \\

		\begin{defi}[Fibré tangent]
		Soit $\mathcal{M}$ une variété lisse de dimension $m$. Alors $\tau_\mathcal{M} : T\mathcal{M} \to \mathcal{M}$ s'appelle le fibré tangent de $\mathcal{M}$. \\
		La variété $T\mathcal{M}$ de dimension $2m$ s'appelle l'espace total et $\mathcal{M}$ la base du fibré tangent. \\
		De plus pour $x\in \mathcal{M}$, la fibre au dessus de $x$, $T_x\mathcal{M}$ est appelé l'espace tangent à $\mathcal{M}$ en $x$. C'est un espace vectoriel de dimension $m$.
		\end{defi}

		\subsubsection{Fibrés vectoriels:}
		Pour $x\in U_\alpha \cap U_\beta$ et $v\in \R^m$, on a
		$$T\varphi_{\beta\alpha}(\phi_\alpha(x), v) = (\phi_\beta(x), d\varphi_{\beta\alpha}(\phi_\alpha(x)).v)$$
		Comme $\varphi_{\beta\alpha}$ est une fonction de transition, c'est une bijection dont l'inverse est lisse donc $d\varphi_{\beta\alpha}(\phi_\alpha(x))$ est une application linéaire inversible. Ainsi, posons
		$$t_{\beta\alpha} = d\varphi_{\beta\alpha}\circ\phi_\alpha : U_\alpha \cap U_\beta \to \GL(\R^m)$$
		On peut alors réecrire 
		$$T\varphi_{\beta\alpha}(\phi_\alpha(x), v) = (\phi_\beta(x), t_{\beta\alpha}(x)v)$$

		\begin{pro}
		Les fonctions $g_{\beta\alpha}$ verifient les propriétés suivantes:
		\be
			\item $t_{\alpha\alpha}(x) = \id, \quad \forall x\in U_\alpha$
			\item $t_{\beta\alpha}(x)^{-1} = t_{\alpha\beta}(x), \quad \forall x\in U_\alpha \cap U_\beta$
			\item $t_{\gamma\alpha}(x) = t_{\gamma\beta}(x)t_{\beta\alpha}(x), \quad \forall x \in U_\alpha \cap U_\beta \cap U_\gamma$
		\ee
		\end{pro}
		\begin{preu}
		On obtient (iii) par composition des applications différentiables, puis (i) en prenant $\alpha = \beta = \gamma$ et (ii) en prenant $\alpha = \gamma$. 
		\end{preu}

		\begin{defi}[Cocycle]
		Soit $\mathcal{M}$ une variété lisse, soit $\{(U_\alpha, \phi_\alpha), \alpha \in A\}$ sont atlas lisse et soit $G$ un groupe de Lie. Un cocycle sur $\mathcal{M}$ à valeur dans $G$ est une famille de fonctions lisses $g_{\beta\alpha} : U_\alpha \cap U_\beta \to G$ telle que $\forall \alpha, \beta \in A$,
		$$g_{\gamma\alpha}(x) = g_{\gamma\beta}(x)g_{\beta\alpha}(x), \quad \forall x \in U_\alpha \cap U_\beta \cap U_\gamma$$ \\
		\end{defi}

		Soit $\mathcal{M}$ une variété lisse de dimension $m$ munie d'un cocycle $g_{\beta\alpha}$ (relatif à l'atlas $\{(U_\alpha, \phi_\alpha), \alpha \in A\}$ de $\mathcal{M}$) à valeur dans un groupe de Lie $G$ et soit $\rho : G \to \GL(\R^k)$ une représentation de $G$. Et définissons $\tilde{g_{\beta\alpha}} : V_{\beta\alpha}\times \R^k \to V_{\alpha\beta} \times \R^k$ par
		$$\tilde{g_{\beta\alpha}}(\phi_\alpha(x), v) = (\phi_\beta(x), \rho(g_{\beta\alpha})v), \quad \forall x\in U_\alpha \cap U_\beta, \quad \forall v\in \R^k$$

		Ainsi de la même façon que pour le fibré tangent, on construit une variété lisse $E$ de dimension $m+k$, une application $\pi : E\to \mathcal{M}$ qui localement correspond à la projection sur la première composante et une structure naturelle d'espace vectoriel sur les fibres $E_x = \pi^{-1}(x)$ au dessus de chaque point $x$ de $\mathcal{M}$. \\

		\begin{defi}[Fibré vectoriel]
		Soit $\mathcal{M}$ une variété lisse muni d'un atlas $\{(U_\alpha, \phi_\alpha), \alpha \in A\}$, d'un cocycle $g_{\beta\alpha}$ à valeur dans un groupe de Lie $G$ et d'une représentation $\rho: G \to \GL(\R^k)$ de $G$. Alors $\pi : E \to \mathcal{M}$ est un fibré vectoriel lisse au dessus de $\mathcal{M}$ avec comme fibre modèle $\R^k$.
		\end{defi}

		\begin{pro}
		Soit $\rho : \GL(R^m) \to \GL(R^k)$ une représentation du groupe de Lie $\GL(\R^m)$, alors $\rho \circ t_{\beta\alpha}$ est un cocycle sur $\mathcal{M}$.
		\end{pro}

		\begin{defi}[Fibré cotangent]
		Le fibré cotangent de $\mathcal{M}$ noté $\tau^*_\mathcal{M} : T^*\mathcal{M} \to \mathcal{M}$ est donné par le cocycle ${}^tt_{\beta\alpha}^{-1}$ à valeur dans $\GL((\R^m)^*)$. 
		\end{defi}

		\begin{defi}[Fibré tensoriel]
		Soient $r,s\in \N$, alors le fibré tensoriel d'ordre $(r,s)$ ($r$ contravariant et $s$ covariant) de $\mathcal{M}$, noté $T^{(r,s)}\mathcal{M}$ est donné par le cocycle $\otimes^r (t_{\beta\alpha}) \otimes^s ({}^tt_{\beta\alpha}^{-1})$ à valeur dans $\GL(\otimes^r (\R^m) \otimes^s (\R^m)^*)$.
		\end{defi}

		\begin{defi}[Section]
		Soit $\pi: E \to \mathcal{M}$ un fibré vectoriel lisse sur $\mathcal{M}$. Une section est une application $s:\mathcal{M} \to E$ telle que $\pi \circ s = \id_\mathcal{M}$ ou de façon équivalente telle que $\forall x\in \mathcal{M}, s(x) \in E_x$. \\
		L'ensemble des sections de $\pi: E \to \mathcal{M}$, noté $\Gamma(E)$ ou $\Gamma(\pi)$ est un espace vetoriel. \\

		On appelle champs de vecteurs les sections du fibré tangent et champs tensoriels les sections des fibrés tensoriels.
		\end{defi}

		\subsubsection{Algèbre exterieure:}

		Soit $V$ un $\R$-espace vectoriel de dimension $n$. Alors l'ensemble $\Sumd_{k\geq 0} \otimes^k V$ munie de l'application 
		$$\begin{array}{ccccc}
			\otimes & : & \otimes^p V\times \otimes^q V & \to & \otimes^{p+q} V \\
 			& & (u_1\otimes ...\otimes u_p, v_1 \otimes ... \otimes v_q) & \mapsto & u_1 \otimes u_p \otimes v_1 \otimes ... \otimes v_q \\
		\end{array}$$
		Est une algèbre dont le produit est $\otimes$. 

		\begin{defi}[Algèbre exterieure]
		Soit $V$ un $\R$-espace vectoriel et soit $k\in \N^*$, on appelle $k$-ème puissance exterieure de $V$, notée $\Lambda^k V$ le quotient de $\otimes^k V$ par l'espace vectoriel engendré par les éléments $v_1 \otimes ... \otimes v_k$ où il existe $i\neq j$ tels que $v_i = v_j$. Aussi on note $\Lambda^0 V = \R$.\\
		On note $v_1 \wedge ... \wedge v_k$ la classe dans ce quotient de l'élément $v_1 \otimes ... \otimes v_k$. \\

		Si $(e_1, ..., e_n)$ est une base de $V$ alors $e_{i_1} \wedge ... \wedge e_{i_k}$ pour $1 \leq i_1 < ... < i_k \leq n$ est une base de $\Lambda^k V$ qui est donc un espace vectoriel de dimension $\bino{k}{n}$. \\
		On note $\Lambda^*V = \Sumd_{k\geq 0} \Lambda^k V$. Le produit $\otimes : \otimes^p V\times \otimes^q V \to \otimes^{p+q} V$ passe au quotient en une application $\wedge : \Lambda^p V \times \Lambda^q V \to \Lambda^{p+q} V$ qui munit $\Lambda^* V$ d'une structure d'algèbre qu'on appelle algèbre exterieure de $V$.
		\end{defi}

		\begin{pro}
		On a les propriétés suivantes:
		\be
			\item L'application $V^k \to \Lambda^k V, (v_1, ..., v_k) \mapsto v_1 \wedge ... \wedge v_k$ est $k$-linéaire alternée.
			\item Si $u \in \Lambda^p V$ et $v \in \Lambda^q V$ alors $u\wedge v = (-1)^{pq} v\wedge u$
		\ee
		\end{pro}

		\begin{pro}
		$\Lambda^k V$ vérifie la proprété universelle suivante: pour toute application $k$-linéaire alternée $f: V^k \to F$ il existe une unique application linéaire $g : \Lambda^k V \to F$ telle que $f(v_1, ..., v_k) = g(v_1\wedge ...\wedge v_k)$.
		\end{pro}
		\begin{preu}
		Comme $f$ est $k$-linéaire par propriété universelle du produit tensoriel, il existe une unique $h: \otimes^k V \to F$ linéaire telle que $f(v_1, ..., v_k) = h(v_1 \otimes ... \otimes v_k)$. Puis comme $f$ est alternée, $h$ descend au quotient en une unique application $g: \Lambda^k V \to F$ linéaire telle que $f(v_1, ..., v_k) = g(v_1\wedge ...\wedge v_k)$.   
		\end{preu}

		\begin{pro}
		Si $V$ est un $\R$-espace vectoriel de dimension fini $n$, alors on a
		$$\otimes^k V^* \simeq (\otimes^k V)^*$$
		et 
		$$\Lambda^k V^* \simeq (\Lambda^k V)^*$$ 
		\end{pro}
		\begin{preu}
		Soit $(e_1, ..., e_n)$ une base de $V$ et notons $(e_1^*, ..., e_n^*)$ la base duale dans $V^*$. Ainsi, on a l'isomorphisme définit par $e_{i_1}^* \otimes ...\otimes e_{i_k}^* \mapsto (e_{i_1} \otimes ...\otimes e_{i_k})^*$ et l'isomorphisme $e_{i_1}^* \wedge ...\wedge e_{i_k}^* \mapsto (e_{i_1} \wedge ... \wedge e_{i_k})^*$
		\end{preu}

		Soit $V, W$ deux espaces vectoriels et soit $f: V \to W$ une application linéaire, alors on définit $\Lambda^k f : \Lambda^k V \to \Lambda^k W$ par
		$$\Lambda^k f (v_1\wedge ... \wedge v_k) = f(v_1) \wedge ... \wedge f(v_k)$$
		et aussi $\Lambda^* f : \Lambda^* V \to \Lambda^* W$ par $\Lambda^* f = \Sumd_{k\geq 0} \Lambda^k f$. \\

		Soit $V$ un $\R$-espace vectoriel euclidien orienté de dimension $n$. Notons $(e_1, ..., e_n)$ une base orthonormée directe de $V$. On pose 
		$$\langle u_1 \wedge ... \wedge u_k | v_1 \wedge ... \wedge v_k \rangle_k = \det (\langle u_i | v_j \rangle)$$
		c'est un produit scalaire sur $\Lambda^k V$ \\

		Notons $\Vol = e_1 \wedge ... \wedge e_n$ appelé élément volume de $\Lambda^n V$. \\

		\begin{defi}[Étoile de Hodge]
		Soit $V$ un $\R$-espace vectoriel euclidien orienté de dimension $n$. Pour $0 \leq k \leq n$, il existe un unique isomorphisme linéaire $\star : \Lambda^k V \to \Lambda^{n-k} V$ appelée étoile de Hodge telle que 
		$$\omega \wedge \rho = \langle \star w | \rho \rangle_{n-k} \Vol, \quad \forall \omega \in \Lambda^k V, \quad \forall \rho \in \Lambda^{n-k} V$$
		\end{defi} 
		\begin{preu}
		Soit $\omega \in \Lambda^k V$ et on pose l'application linéaire $\rho \in \Lambda^{n-k} V \mapsto \omega \wedge \rho \in \Lambda^n V$. Comme $\Lambda^n V$ est de dimension 1, on a $\omega \wedge \rho = c(\rho)\Vol$. Alors $c : \Lambda^{n-k} V \to \R$ est une forme linéaire donc se représente par un unique $\star\omega \in \Lambda^{n-k} V$.
		\end{preu}

		\subsubsection{Formes différentielles:}

		Soit $k\in \N^*$, on définit la $k$-ème puissance du produit fibré du fibré tangent $T\mathcal{M}$ avec lui-même:
		$$T^{\diamond k}\mathcal{M} = \{ (v_1, ..., v_k)\in (T\mathcal{M})^k | \tau_\mathcal{M}(v_1) = ... = \tau_\mathcal{M}(v_k)\}$$

		\begin{defi}[Forme différentielle]
		Une $k$-forme differentielle sur $\mathcal{M}$ est une application lisse $T^{\diamond k}\mathcal{M} \to \R$ telle que la restriction a chaque fibre soit une forme $k$-linéaire alternée. On note $\Omega^k\mathcal{M}$ l'espace vectoriel des $k$-formes différentielles sur $\mathcal{M}$. Aussi on note $\Omega^0\mathcal{M} = \mathscr{C}^\infty(\mathcal{M})$.
		\end{defi}

		\begin{pro}
		Considérons la représentation $\rho_k : \GL((\R^m)^*) \to \GL(\Lambda^k (\R^m)^*), f \mapsto \Lambda^k f$ et notons $\rho_k(\tau^*_\mathcal{M})$ le fibré vectoriel induit par la représentation $\rho_k$ et par le fibré cotangent $\tau^*_\mathcal{M}$ de $\mathcal{M}$. Alors,
		$$\Omega^k\mathcal{M} \simeq \Gamma(\rho_k(\tau^*_\mathcal{M}))$$
		\end{pro}

		On dispose de l'application $\wedge : \Omega^p\mathcal{M} \times \Omega^q\mathcal{M} \to \Omega^{p+q}\mathcal{M}$ appelé produit extérieur qui est induit par le produit extérieur sur les espaces $\Lambda^k(\R^m)^*$. Ainsi, l'espace $\Omega^*\mathcal{M} = \Sumd_{k\geq 0} \Omega^k\mathcal{M}$ muni du produit exterieur $\wedge$ est une algèbre. \\

		\begin{defi}
		Soient $\mathcal{M}, \mathcal{N}$ deux variétés lisse, $f:\mathcal{N} \to \mathcal{M}$ une application lisse et $\eta \in \Omega^k\mathcal{M}$. Alors $f^*\eta = \eta (Tf, ..., Tf) : T^{\diamond k}\mathcal{N} \to \R$ est une $k$-forme différentielle sur $\mathcal{N}$ appelée la forme tirée en arrière par $f$ de $\eta$. \\
		\end{defi}

		\begin{defi}[Différentielle extérieure]
		Soit $(U, \phi)$ une carte de $\mathcal{M}$ et soit $x\in U$ et notons $\phi = (x^1, ..., x^m)$ et notons $(\partial_1, ..., \partial_m)$ la base de $T_x\mathcal{M}$ image reciproque de la base canonique de $\R^m$ par $T_x\phi$ et notons $(dx^1, ..., dx^m)$ la base duale de $(\partial_1, ..., \partial_m)$ dans $(T_x\mathcal{M})^*$. Pour $f\in \Omega^0\mathcal{M} = \mathscr{C}^\infty(\mathcal{M})$ on définit l'application linéaire $d : \Omega^0\mathcal{M} \to \Omega^1\mathcal{M}$ par 
		$$df = \frac{\partial f \circ \phi^{-1}}{\partial x^1}dx^1 +...+ \frac{\partial f \circ \phi^{-1}}{\partial x^m}dx^m$$
		En particulier on a $d(x^k) = dx^k$. \\
		Puis pour $\omega \in \Omega^k\mathcal{M}$ une $k$-forme sur $\mathcal{M}$ s'écrivant $\omega = f dx^{i_1} \wedge ... \wedge dx^{i_k}$ alors on définit l'application linéaire $d: \Omega^k\mathcal{M} \to \Omega^{k+1}\mathcal{M}$ par 
		$$d\omega = df \wedge dx^{i_1} \wedge ... \wedge dx^{i_k}$$
		On appelle les application $d: \Omega^k\mathcal{M} \to \Omega^{k+1}\mathcal{M}$ la différentielle extérieure sur $\mathcal{M}$.
		\end{defi}

		\begin{pro}
		La dérivée exterieure $d: \Omega^k\mathcal{M} \to \Omega^{k+1}\mathcal{M}$ vérifie les propriétés suivantes :
		\be
			\item $d$ est linéaire
			\item $d^2 = 0$ 
			\item Pour $\omega_1 \in \Omega^p\mathcal{M}$ et $\omega_2 \in \Omega^q\mathcal{M}$, $\quad d(\omega_1 \wedge \omega_2) = d\omega_1 \wedge \omega_2 + (-1)^p\omega_1 \wedge d\omega_2$
			\item Pour $f\in \Omega^0\mathcal{M}$ et pour tout champs de vecteurs $X$ sur $\mathcal{M}$ on a $df(X) = fX$
			\item Pour $f:\mathcal{N} \to \mathcal{M}$ lisse et pour $\omega \in \Omega^k\mathcal{M}$ on a $f^*(d\omega) = d(f^*\omega)$
			\item Pour tout champs de vecteurs $X,Y$ sur $\mathcal{M}$ et pour toute $1$-forme $\omega \in \Omega^1\mathcal{M}$ on a $d\omega(X,Y) = X(\omega(Y)) - Y(\omega(X)) - \omega([X, Y])$
		\ee
		\end{pro}

		\begin{defi}[Cohomologie de Rham]
		Soit $\mathcal{M}$ une variété lisse, pour $k\in \N^*$ on définit la $k$-ème cohomologie de Rham l'espace quotient 
		$$H_{dr}^k\mathcal{M} = \frac{\Ker d_{k+1}}{\Ima d_k}$$
		où $d_k = d: \Omega^k\mathcal{M} \to \Omega^{k+1}\mathcal{M}$.
		\end{defi}

		\begin{exe}
		Soit $U$ est un ouvert de $\R^3$, on identifie $(\R^3)^*$ et $\R^3$. Alors pour $f \in \Omega^0U = \mathscr{C}^\infty(U)$ on a:
		$$df = \frac{\partial f}{\partial x} dx + \frac{\partial f}{\partial y} dy + \frac{\partial f}{\partial z} dz = \grad f$$
		Soit $\omega = f_x dx + f_y dy + f_z dz \in \Omega^1U$, alors on a
		$$d\omega = \left(\frac{\partial f_z}{\partial y} - \frac{\partial f_y}{\partial z}\right)dy\wedge dz + \left(\frac{\partial f_x}{\partial z} - \frac{\partial f_z}{\partial x}\right)dz\wedge dx + \left(\frac{\partial f_y}{\partial x} - \frac{\partial f_x}{\partial y}\right)dx\wedge dy$$
		or $\star (dy\wedge dz) = dx$, $\quad\star (dz\wedge dx) = dy$ et $\quad\star (dx\wedge dy) = dz$ et on a 
		$$\star d\omega = \rot f$$ 
		Soit $\omega = f_x dy\wedge dz + f_y dz\wedge dx + f_z dx \wedge dy \in \Omega^2U$, alors on a 
		$$d\omega = \left(\frac{\partial f_x}{\partial x} + \frac{\partial f_y}{\partial y} + \frac{\partial f_z}{\partial z}\right) dx\wedge dy\wedge dz$$
		or $\star (dx\wedge dy\wedge dz) = 1$ et
		$$\star d\omega = \Div f$$
		En utilisant $d^2 = 0$ on a immediatement 
		$$\rot(\grad f) = 0 $$
		et 
		$$\Div(\rot f) = 0$$
		\end{exe}

		\subsubsection{Flot local d'un champs de vecteurs:}

		\begin{defi}[Flot local d'un champs de vecteurs]
		Soit $X$ un champs de vecteurs sur $\mathcal{M}$. Un flot local de $X$ est une application $\phi : \Omega \to \mathcal{M}$ lisse notée $(t,x) \to \phi_t(x)$ où $\Omega$ est un voisinage ouvert de $\{0\}\times \mathcal{M}$ dans $\R \times \mathcal{M}$ tel que pour tout $x\in \mathcal{M}$, $\Omega \cap \R \times \{x\}$ soit connexe, et qui vérifie les deux propriétés suivantes: 
		$$\frac{d\phi_t(x)}{dt}_{|t=s} = X(\phi_s(x)) \quad \forall (s, x)\in \Omega$$
		$$\phi_0(x) = x \quad \forall x\in \mathcal{M}$$
		Il est dit complet si $\Omega = \R \times \mathcal{M}$. \\
		Il est dit maximal s'il n'existe pas de prolongement strict. Autrement dit tel que s'il existe $\phi' : \Omega' \to \mathcal{M}$ un flot local de $X$ tel que $\Omega \subset \Omega'$ et $\phi'|_{\Omega} = \phi$ alors $\Omega = \Omega'$ et $\phi = \phi'$. Dans ce cas il est alors unique.
		\end{defi}

		\begin{defi}[Courbe integrale]
		Soit $X$ un champs de vecteurs sur $\mathcal{M}$. Une courbe integrale de $X$ est une application $\gamma : I \to \mathcal{M}$ telle que $\dot\gamma = X \circ \gamma$ sur $I$. \\
		Elle est dite maximale s'il n'existe pas de prolongement strict. Autrement dit tel que s'il existe une autre courbe integrale $c: J \to \mathcal{M}$ tel que $I \subset J$ et $c|_I = \gamma$ alors $I = J$ et $c = \gamma$.
		\end{defi}

		\begin{pro}
		Notons $\supp(X) = \overline{\{x\in \mathcal{M} | X(x) \neq 0\}}$. Alors 
		\be
			\item S'il existe $\varepsilon > 0$ tel que $]-2\varepsilon; 2\varepsilon[\times \mathcal{M} \subset \Omega$ alors $\phi_t^X$ est complet.
			\item Si $\supp(X)$ est compact alors $\phi_t^X$ est complet.
		\ee
		\end{pro}

		\begin{pro}
		Un champs de vecteurs $X$ sur $\mathcal{M}$ admet un unique flot maximal $\phi_t^X : \Omega \to \mathcal{M}$ qui vérifie de plus les propriétés suivantes:
		\be
			\item $\forall x_0\in \mathcal{M}$, l'unique courbe integrale maximale de $X$ passant par $x_0$ au temps $t=0$ est la courbe $I_{x_0} = \{t\in \R | (t, x_0) \in \Omega\} \to \mathcal{M}$ définie par
			$$t \mapsto \phi_t^X(x_0)$$

			\item $\forall (s, x)\in \Omega$, on a $(t, \phi_s^X(x)) \in \Omega \iff (t+s, x) \in \Omega$ et dans ce cas
			$$\phi_t^X \circ \phi_s^X(x) = \phi_{t+s}^X(x)$$

			\item $\forall t\in \R$ l'application $\phi_t^X : U_t = \{x\in \mathcal{M} | (t, x) \in \Omega \} \to \mathcal{M}$ est un $\mathscr{C}^\infty$-difféomorphisme local. Si de plus $X$ est complet alors $(\phi_t^X)_{t\in\R}$ est un groupe à un paramètre de $\mathscr{C}^\infty$-difféomorphismes de $\mathcal{M}$. 

			\item $\forall (t,x) \in \Omega$, le flot local préserve le champs de vecteurs, c'est à dire 
			$$T_x\phi_t^X(X(x)) = X(\phi_t^X(x))$$
		\ee
		\end{pro}

		\subsubsection{Dérivation de Lie:}

		\begin{defi}[Dérivation]
		Une dérivation de $\mathcal{M}$ est une application $\R$-linéaire $\delta : \mathscr{C}^\infty(\mathcal{M}) \to \mathscr{C}^\infty(\mathcal{M})$ telle que pour tout $f, g \in \mathscr{C}^\infty(\mathcal{M})$ on ait
		$$\delta(fg) = f\delta(g) + g\delta(f)$$
		On note $\mathscr{D}(\mathcal{M})$ l'ensemble des dérivations sur $\mathcal{M}$. C'est un $\R$-espace vectoriel et c'est aussi un $\mathscr{C}^\infty(\mathcal{M})$-module.
		\end{defi}

		\begin{defi}[Dérivation de Lie]
		Soit $X$ un champs de vecteurs sur $\mathcal{M}$ alors l'application $\mathscr{L}_X: \mathscr{C}^\infty(\mathcal{M}) \to \mathscr{C}^\infty(\mathcal{M})$, appelée la dérivation de Lie associée à $X$ et notée parfois $X(f) = \mathscr{L}_X(f)$, est définie par 
		$$\mathscr{L}_X(f): x\to T_xf(X(x))$$
		Le théorème de dérivation des fonctions composées montre que si $\phi$ est un flot local sur $X$ alors
		$$\mathscr{L}_X(f) = \frac{d}{dt}_{|t = 0}f\circ\phi_t$$
		\end{defi}
